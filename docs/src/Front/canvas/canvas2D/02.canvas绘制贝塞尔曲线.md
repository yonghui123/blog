### 绘制二次贝塞尔曲线

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
  </head>
  <body>
    <script>
      (() => {
        const canvas = document.createElement("canvas");
        const ctx = canvas.getContext("2d");
        canvas.width = 800;
        canvas.height = 600;
        document.body.appendChild(canvas);

        ctx.beginPath();
        ctx.moveTo(50, 50);
        ctx.lineTo(50, 200);
        ctx.moveTo(250, 50);
        ctx.lineTo(250, 200);
        ctx.strokeStyle = "#666666";
        ctx.lineWidth = 2;
        ctx.stroke();

        ctx.beginPath();
        ctx.moveTo(50, 100);
        ctx.lineTo(250, 100);
        ctx.setLineDash([200]);
        ctx.lineDashOffset = 200;
        ctx.strokeStyle = "#ff0000";
        ctx.lineWidth = 10;
        ctx.stroke();

        function move() {
          ctx.clearRect(50, 95, 200, 10);
          ctx.beginPath();
          ctx.moveTo(50, 100);
          ctx.lineTo(250, 100);
          ctx.setLineDash([200]);
          ctx.lineDashOffset -= 2;
          if (ctx.lineDashOffset < -200) {
            ctx.lineDashOffset = 200;
          }
          ctx.stroke();
          requestAnimationFrame(move);
        }
        requestAnimationFrame(move);
      })();

      (() => {
        const canvas = document.createElement("canvas");
        const ctx = canvas.getContext("2d");
        canvas.width = 800;
        canvas.height = 600;
        document.body.appendChild(canvas);
        ctx.strokeStyle = "#666666";
        ctx.lineWidth = 10;
        ctx.beginPath();
        ctx.moveTo(50, 50);
        ctx.lineTo(50, 200);
        ctx.stroke();
        ctx.lineTo(250, 200);
        ctx.closePath();
        ctx.stroke();
        ctx.fill();
      })();

      (() => {
        const canvas = document.createElement("canvas");
        const ctx = canvas.getContext("2d");
        canvas.width = 800;
        canvas.height = 800;
        document.body.appendChild(canvas);
        // 绘制圆弧
        ctx.arc(100, 100, 50, 0, Math.PI, true);
        ctx.strokeStyle = "#666666";
        ctx.lineWidth = 10;
        ctx.stroke();

        // 绘制椭圆
        ctx.beginPath();
        ctx.ellipse(100, 200, 50, 25, Math.PI / 4, 0, Math.PI, true);
        ctx.strokeStyle = "#666666";
        ctx.lineWidth = 10;
        ctx.stroke();
      })();
    </script>
  </body>
</html>
```

使用canvas绘制二次贝塞尔曲线，可以使用`ctx.quadraticCurveTo(cpx, cpy, endX, endY)`：
  - cpx, cpy 表示控制点
  - endX, endY 表示终点

```js
// 绘制贝塞尔曲线
ctx.beginPath();
ctx.moveTo(50, 50);
ctx.quadraticCurveTo(150, 0, 250, 50);
```

线段的起点是moveTo移动到的点，或者上一次绘制的终点坐标 

### 绘制三次贝塞尔曲线

```html
<!DOCTYPE html>
<html>
  <head>
    <title>贝塞尔曲线 GIF 生成</title>
    <script src="./gif.js"></script>
  </head>
  <body>
    <canvas id="canvas" width="400" height="300"></canvas>
    <div>
      <button onclick="startRecording()" id="startBtn">开始录制</button>
      <button onclick="stopRecording()" disabled id="stopBtn">停止并保存</button>
    </div>

    <script>
      const canvas = document.getElementById("canvas");
      const ctx = canvas.getContext("2d");
      let isRecording = false;
      let gif = null;
      let animationId = null;

      // 定义坐标点
      const points = {
        p1: { x: 50, y: 150, color: "red" },
        p2: { x: 300, y: 150, color: "red" },
        c1: { x: 150, y: 50, color: "blue" },
      };

      // 动画参数
      let t = 0;

      function lerp(a, b, t) {
        return {
          x: a.x + (b.x - a.x) * t,
          y: a.y + (b.y - a.y) * t,
        };
      }

      function drawPoint(point, label) {
        ctx.beginPath();
        ctx.arc(point.x, point.y, 5, 0, Math.PI * 2);
        ctx.fillStyle = point.color;
        ctx.fill();
        ctx.fillStyle = "black";
        ctx.font = "12px Arial";
        ctx.fillText(label, point.x + 8, point.y - 8);
      }

      function drawFrame() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // 绘制控制线
        ctx.setLineDash([5, 5]);
        ctx.beginPath();
        ctx.moveTo(points.p1.x, points.p1.y);
        ctx.lineTo(points.c1.x, points.c1.y);
        ctx.lineTo(points.p2.x, points.p2.y);
        ctx.strokeStyle = "#999";
        ctx.stroke();
        ctx.setLineDash([]);

        Object.entries(points).forEach(([label, point]) => drawPoint(point, label));

        const t1 = lerp(points.p1, points.c1, t);
        const t2 = lerp(points.c1, points.p2, t);
        const t3 = lerp(t1, t2, t);

        ctx.beginPath();
        ctx.moveTo(t1.x, t1.y);
        ctx.lineTo(t2.x, t2.y);
        ctx.strokeStyle = "green";
        ctx.stroke();
        drawPoint({ ...t1, color: "green" }, "t1");
        drawPoint({ ...t2, color: "green" }, "t2");
        drawPoint({ ...t3, color: "black" }, "t3");

        ctx.beginPath();
        ctx.moveTo(points.p1.x, points.p1.y);
        for (let i = 0; i <= t; i += 0.01) {
          const ti1 = lerp(points.p1, points.c1, i);
          const ti2 = lerp(points.c1, points.p2, i);
          const ti3 = lerp(ti1, ti2, i);
          ctx.lineTo(ti3.x, ti3.y);
        }
        ctx.strokeStyle = "black";
        ctx.stroke();

        if (isRecording) {
          gif.addFrame(ctx, { copy: true, delay: 20 });
        }

        t += 0.01;

        // 重要修改：当到达终点时自动停止
        if (t >= 1) {
          t = 1; // 保持终点状态
          stopRecording();
          return; // 停止继续绘制
        }

        animationId = requestAnimationFrame(drawFrame);
      }

      function startRecording() {
        if (!isRecording) {
          // 重置状态
          cancelAnimationFrame(animationId);
          t = 0;

          // 初始化GIF
          gif = new GIF({
            workers: 2,
            quality: 10,
            width: canvas.width,
            height: canvas.height,
          });

          gif.on("finished", (blob) => {
            const url = URL.createObjectURL(blob);
            const a = document.createElement("a");
            a.href = url;
            a.download = `bezier_${Date.now()}.gif`;
            a.click();
            resetAnimation();
          });

          // 更新UI状态
          document.getElementById("startBtn").disabled = true;
          document.getElementById("stopBtn").disabled = false;
          isRecording = true;

          // 开始绘制
          drawFrame();
        }
      }

      function stopRecording() {
        if (isRecording) {
          gif.render();
          isRecording = false;
        }
      }

      function resetAnimation() {
        // 重置所有状态
        cancelAnimationFrame(animationId);
        t = 0;
        document.getElementById("startBtn").disabled = false;
        document.getElementById("stopBtn").disabled = true;
        // 绘制初始帧
        drawFrame();
      }

      // 初始化绘制
      drawFrame();
    </script>
  </body>
</html>
```

canvas中绘制三次贝塞尔曲线，可以使用`ctx.bezierCurveTo(cpx1, cpy1, cpx2, cpy2, endX, endY)`来绘制
  - cpx1,cpy1 控制点1
  - cpx2,cpy2 控制点2
  - endX,endY 终点
  
起点是上一次绘制的终点，或者moveTo移动到的位置点
